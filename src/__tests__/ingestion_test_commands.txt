//Plain Text(TXT) Ingestion
curl -X POST http://localhost:3000/ingest \
  -F "file=@C:\Users\Admin\Desktop\notes.txt"

//Word Document(DOCX) Ingestion
curl -X POST http://localhost:3000/ingest \
  -F "file=@C:\Users\Admin\Documents\report.docx"


//URL Ingestion(JSON)
curl -X POST http://localhost:3000/ingest \
  -H "Content-Type: application/json" \
  -d '{"url": "https://en.wikipedia.org/wiki/Large_language_model"}'

    "metadata": { "author": "admin", "priority": "high" }
